\subsection{n的顶化}

\begin{frame}{\insertsubsection  ~~条件 }
  \begin{enumerate}
    \item
      肯定条件：
      
      词内n前有r, ṛ, ṝ或ṣ。
    \item
      肯定条件：
      
      n后是n, m, y, v或者元音。
    \item
      否定条件：
      
      r, ṛ, ṝ或ṣ和n之间有齿音、腭音、顶音或者s和ś。
  \end{enumerate}
\end{frame}
