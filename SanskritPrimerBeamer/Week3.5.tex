\subsection{独立依格}
\begin{frame}{\insertsection ~~\insertsubsection}
  \small
  \begin{itemize}
    \item 独立依格 (Locative Absolute)：
    \begin{itemize}
      \item 名词/代词 + 分词，都用\important{依格}
      \item 翻译为时间从句：\\
        (when / while / after)
      \item 名词作主语，分词作谓语
    \end{itemize}
    \item 分词类型：
    \begin{itemize}
      \item 过去分词 → when/after\\
      \item 现在分词 → while/as
      \item 完成时分词等皆可
    \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}{\insertsubsection~~例句}
  \small
  \begin{indentlist}
    \item \sktsentence{gate tasmin narā upāviśan}\\
      \notsoimportant{当他离开后，众人坐下了}\\
      \notsoimportant{(lit. 'at him having gone')}
    \item \sktsentence{udyati sūrye vanaṃ praviśati}\\
      \notsoimportant{日出时，他进入森林}\\
      \notsoimportant{(ud-\verbroot{i(2)} 'to go up, rise')}\\
    \item \sktsentence{evaṃ gate}\\
      \notsoimportant{在这样的情况下（无人称结构）}\\
  \end{indentlist}
\end{frame}

\subsection{独立属格}
\begin{frame}{\insertsection ~~\insertsubsection}
  \small
  \begin{itemize}
    \item 独立属格 (Genitive Absolute)：
    \begin{itemize}
      \item 名词/代词 + 分词，都用\important{属格}
      \item 比独立依格\veryimportant{少见}
      \item 可表时间（when/while）\\
        或\important{让步}（although）
    \end{itemize}
  \end{itemize}

  \begin{indentlist}
    \item \sktsentence{maitrāvaruṇiḥ samudramapibat ...}\\
      \sktsentence{sarvalokasya paśyataḥ}\\
      \notsoimportant{密多罗和伐楼那之子喝干了海水，}\\
      \notsoimportant{（当时/尽管）全世界都在看}\\
      \hfill\notsoimportant{(\emph{Mahābhārata} 3.103.3)}
  \end{indentlist}
\end{frame}

\subsection{概述}
\begin{frame}{ena- \insertsubsection}
  \small
  \begin{itemize}
    \item 指示代词 \nounstem{ena} 'this'
    \item 用于\important{无强调}的指代
    \item \veryimportant{不完全变化} (defective)\\
    \item 仅有部分形式存在：
    \begin{itemize}
      \item 业格（单、双、复）
      \item 具格（单数）
      \item 属依格（双数，仅阳/中性）
    \end{itemize}
  \end{itemize}
\end{frame}
